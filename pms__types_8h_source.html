149 // normalize
150 void normalize() {
151 if (x == 0.0f && y == 0.0f && z == 0.0f) {
152 return;
153 }
154 else {
155 const float32 sq = x * x + y * y + z * z;
156 const float32 sqf = sqrt(sq);
157 x /= sqf; y /= sqf; z /= sqf;
158 }
159 }

162 // operator +

164 return PVector3f(x + v.x, y + v.y, z + v.z);
165 }

168 return PVector3f(x - v.x, y - v.y, z - v.z);
169 }
170 // operator -t
171 PVector3f operator-() const {

173 }
174 // operator =

176 if (this == &v) {

180 x = v.x;
181 y = v.y;
182 z = v.z;
183 return *this;

186 // operator ==

189 }
190 // operator !=

192 return (x != v.x) || (y != v.y) || (z != v.z);
