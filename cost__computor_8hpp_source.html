130 inline float32 Compute(const sint32& x, const sint32& y, const float32& d) override
131 {
132 const float32 xr = x - d;
133 if (xr < 0.0f || xr >= static_cast<float32>(width_)) {
134 return (1 - alpha_) * tau_col_ + alpha_ * tau_grad_;
135 }
136 // 颜色空间距离
137 const auto col_p = GetColor(img_left_, x, y);
138 const auto col_q = GetColor(img_right_, xr, y);
139 const auto dc = std::min(abs(col_p.b - col_q.x) + abs(col_p.g - col_q.y) + abs(col_p.r - col_q.z), tau_col_);
140
141 // 梯度空间距离
142 const auto grad_p = GetGradient(grad_left_, x, y);
143 const auto grad_q = GetGradient(grad_right_, xr, y);
144 const auto dg = std::min(abs(grad_p.x - grad_q.x)+ abs(grad_p.y - grad_q.y), tau_grad_);
145
146 // 代价值
147 return (1 - alpha_) * dc + alpha_ * dg;
148 }

185 inline float32 ComputeA(const sint32& x, const sint32& y, const DisparityPlane& p) const
186 {
187 const auto pat = patch_size_ / 2;
188 const auto& col_p = GetColor(img_left_, x, y);
189 float32 cost = 0.0f;
190 for (sint32 r = -pat; r <= pat; r++) {
191 const sint32 yr = y + r;
192 for (sint32 c = -pat; c <= pat; c++) {
193 const sint32 xc = x + c;
194 if (yr < 0 || yr > height_ - 1 || xc < 0 || xc > width_ - 1) {
195 continue;
196 }
197 // 计算视差值
198 const float32 d = p.to_disparity(xc,yr);
199 if (d < min_disp_ || d > max_disp_) {
200 cost += COST_PUNISH;
201 continue;
202 }
203
204 // 计算权值
205 const auto& col_q = GetColor(img_left_, xc, yr);
206 const auto dc = abs(col_p.r - col_q.r) + abs(col_p.g - col_q.g) + abs(col_p.b - col_q.b);
207#ifdef USE_FAST_EXP
208 const auto w = fast_exp(double(-dc / gamma_));
209#else
210 const auto w = exp(-dc / gamma_);
211#endif
212
213 // 聚合代价
214 const auto grad_q = GetGradient(grad_left_, xc, yr);
215 cost += w * Compute(col_q, grad_q, xc, yr, d);
216 }
217 }
218 return cost;
219 }

241 inline PVector3f GetColor(const uint8* img_data, const float32& x,const sint32& y) const
242 {
243 float32 col[3];
244 const auto x1 = static_cast<sint32>(x);
245 const sint32 x2 = x1 + 1;
246 const float32 ofs = x - x1;
247
248 for (sint32 n = 0; n < 3; n++) {
249 const auto& g1 = img_data[y * width_ * 3 + 3 * x1 + n];
250 const auto& g2 = (x2 < width_) ? img_data[y * width_ * 3 + 3 * x2 + n] : g1;
251 col[n] = (1 - ofs) * g1 + ofs * g2;
252 }
253
254 return { col[0], col[1], col[2] };
255 }

276 inline PVector2f GetGradient(const PGradient* grad_data, const float32& x, const sint32& y) const
277 {
278 const auto x1 = static_cast<sint32>(x);
279 const sint32 x2 = x1 + 1;
280 const float32 ofs = x - x1;

282 const auto& g1 = grad_data[y * width_ + x1];

285 return { (1 - ofs) * g1.x + ofs * g2.x,(1 - ofs) * g1.y + ofs * g2.y };
